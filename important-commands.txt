#!/usr/bin/env bash - не во всех системах файл bash/ находится по ссылке /bin/bash. Для лучшей совместимости необходимо использовать указанную команду

set -Eeuo pipefail - уловитель ошибок в коде. Для близкого ознакомления использовать команду set --help

script_dir=$(cd "$(dirname "${BASH_SOURCE[0]}")" &>/dev/null && pwd -P) - Получить путь к каталогу, в котором находится скрипт, внутри скрипта.
